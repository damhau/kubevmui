# Phase 1 MVP Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build a functional single-cluster KubeVirt management UI with VM lifecycle, console access, templates, network profiles, and disk management — a control plane that abstracts Kubernetes behind operator-friendly concepts.

**Architecture:** FastAPI backend exposing our own resource model (VM, Disk, NetworkProfile, Template) with a service layer translating to/from K8s CRDs. React frontend consuming the auto-generated OpenAPI client. Two parallel tracks: backend and frontend, synchronized via the OpenAPI spec.

**Tech Stack:** Python 3.12 + FastAPI + uv (backend), React 18 + TypeScript + Vite + shadcn/ui + TanStack Query + Zustand (frontend), Docker Compose (dev), noVNC + xterm.js (consoles)

```python
# backend/app/core/cluster_manager.py
from dataclasses import dataclass, field

from kubernetes import client, config


@dataclass
class ClusterInfo:
    name: str
    api_client: client.ApiClient | None = None
    is_local: bool = False
    connected: bool = False


class ClusterManager:
    def __init__(self, kubeconfig_path: str | None = None, in_cluster: bool = True):
        self._clusters: dict[str, ClusterInfo] = {}
        self._setup_local_cluster(kubeconfig_path, in_cluster)

    def _setup_local_cluster(self, kubeconfig_path: str | None, in_cluster: bool):
        api_client = None
        connected = False
        try:
            if in_cluster:
                config.load_incluster_config()
                api_client = client.ApiClient()
                connected = True
            elif kubeconfig_path:
                api_client = config.new_client_from_config(config_file=kubeconfig_path)
                connected = True
        except Exception:
            pass

        self._clusters["local"] = ClusterInfo(
            name="local",
            api_client=api_client,
            is_local=True,
            connected=connected,
        )

    def list_clusters(self) -> list[ClusterInfo]:
        return list(self._clusters.values())

    def get_cluster(self, name: str) -> ClusterInfo | None:
        return self._clusters.get(name)

    def get_api_client(self, cluster_name: str) -> client.ApiClient | None:
        cluster = self._clusters.get(cluster_name)
        if cluster and cluster.api_client:
            return cluster.api_client
        return None
```

```python
# backend/app/core/k8s_client.py
from kubernetes import client
from kubernetes.client import ApiException


class KubeVirtClient:
    """Wraps the Kubernetes API client for KubeVirt CRD operations."""

    KUBEVIRT_API_GROUP = "kubevirt.io"
    KUBEVIRT_API_VERSION = "v1"
    SUBRESOURCE_API_GROUP = "subresources.kubevirt.io"

    def __init__(self, api_client: client.ApiClient):
        self.api_client = api_client
        self.custom_api = client.CustomObjectsApi(api_client)
        self.core_api = client.CoreV1Api(api_client)

    def list_vms(self, namespace: str) -> list[dict]:
        result = self.custom_api.list_namespaced_custom_object(
            group=self.KUBEVIRT_API_GROUP,
            version=self.KUBEVIRT_API_VERSION,
            namespace=namespace,
            plural="virtualmachines",
        )
        return result.get("items", [])

    def get_vm(self, namespace: str, name: str) -> dict | None:
        try:
            return self.custom_api.get_namespaced_custom_object(
                group=self.KUBEVIRT_API_GROUP,
                version=self.KUBEVIRT_API_VERSION,
                namespace=namespace,
                plural="virtualmachines",
                name=name,
            )
        except ApiException as e:
            if e.status == 404:
                return None
            raise

    def create_vm(self, namespace: str, body: dict) -> dict:
        return self.custom_api.create_namespaced_custom_object(
            group=self.KUBEVIRT_API_GROUP,
            version=self.KUBEVIRT_API_VERSION,
            namespace=namespace,
            plural="virtualmachines",
            body=body,
        )

    def delete_vm(self, namespace: str, name: str) -> None:
        self.custom_api.delete_namespaced_custom_object(
            group=self.KUBEVIRT_API_GROUP,
            version=self.KUBEVIRT_API_VERSION,
            namespace=namespace,
            plural="virtualmachines",
            name=name,
        )

    def patch_vm(self, namespace: str, name: str, body: dict) -> dict:
        return self.custom_api.patch_namespaced_custom_object(
            group=self.KUBEVIRT_API_GROUP,
            version=self.KUBEVIRT_API_VERSION,
            namespace=namespace,
            plural="virtualmachines",
            name=name,
            body=body,
        )

    def vm_action(self, namespace: str, name: str, action: str) -> dict:
        """Trigger a VM subresource action (start, stop, restart, pause, unpause)."""
        path_params = {
            "namespace": namespace,
            "name": name,
        }
        resource_path = (
            f"/apis/{self.SUBRESOURCE_API_GROUP}/{self.KUBEVIRT_API_VERSION}"
            f"/namespaces/{namespace}/virtualmachines/{name}/{action}"
        )
        return self.api_client.call_api(
            resource_path, "PUT",
            path_params=path_params,
            body={},
            response_type="object",
            _return_http_data_only=True,
        )

    def get_vmi(self, namespace: str, name: str) -> dict | None:
        """Get VirtualMachineInstance (running state)."""
        try:
            return self.custom_api.get_namespaced_custom_object(
                group=self.KUBEVIRT_API_GROUP,
                version=self.KUBEVIRT_API_VERSION,
                namespace=namespace,
                plural="virtualmachineinstances",
                name=name,
            )
        except ApiException as e:
            if e.status == 404:
                return None
            raise

    def list_vmis(self, namespace: str) -> list[dict]:
        result = self.custom_api.list_namespaced_custom_object(
            group=self.KUBEVIRT_API_GROUP,
            version=self.KUBEVIRT_API_VERSION,
            namespace=namespace,
            plural="virtualmachineinstances",
        )
        return result.get("items", [])

    def list_namespaces(self) -> list[str]:
        result = self.core_api.list_namespace()
        return [ns.metadata.name for ns in result.items]

    def list_nodes(self) -> list[dict]:
        result = self.core_api.list_node()
        nodes = []
        for node in result.items:
            nodes.append({
                "name": node.metadata.name,
                "status": "Ready" if any(
                    c.type == "Ready" and c.status == "True"
                    for c in node.status.conditions or []
                ) else "NotReady",
                "roles": [
                    k.replace("node-role.kubernetes.io/", "")
                    for k in (node.metadata.labels or {})
                    if k.startswith("node-role.kubernetes.io/")
                ],
                "cpu_capacity": node.status.capacity.get("cpu", "0"),
                "memory_capacity": node.status.capacity.get("memory", "0"),
                "cpu_allocatable": node.status.allocatable.get("cpu", "0"),
                "memory_allocatable": node.status.allocatable.get("memory", "0"),
            })
        return nodes
```

```python
# backend/app/auth/k8s_token.py
from kubernetes import client
from kubernetes.client import ApiException

from app.models.auth import UserInfo


async def validate_k8s_token(token: str, api_client: client.ApiClient | None) -> UserInfo:
    """Validate a K8s bearer token by calling the TokenReview API."""
    if not api_client:
        return UserInfo(username="anonymous", authenticated=False)

    auth_api = client.AuthenticationV1Api(api_client)
    try:
        review = client.V1TokenReview(
            spec=client.V1TokenReviewSpec(token=token)
        )
        result = auth_api.create_token_review(review)
        if result.status.authenticated:
            return UserInfo(
                username=result.status.user.username,
                groups=result.status.user.groups or [],
                authenticated=True,
            )
    except ApiException:
        pass

    return UserInfo(username="anonymous", authenticated=False)
```

```python
async def get_current_user(request: Request) -> UserInfo:
    token = None

    if auth_header and auth_header.startswith("Bearer "):
        token = auth_header[7:]

    cookie_token = request.cookies.get("kubevmui_token")
    if not token and cookie_token:
        token = cookie_token

    if not token:
        raise HTTPException(status_code=401, detail="Not authenticated")

    from app.auth.k8s_token import validate_k8s_token
    cm: ClusterManager = request.app.state.cluster_manager
    api_client = cm.get_api_client("local")
    user = await validate_k8s_token(token, api_client)

    if not user.authenticated:
        raise HTTPException(status_code=401, detail="Invalid token")

    return user
```

```python
# backend/app/services/vm_service.py
from app.core.k8s_client import KubeVirtClient
from app.models.common import HealthStatus, VMStatus
from app.models.vm import VM, VMCompute, VMCreate, VMDiskRef, VMList, VMNetworkRef


class VMService:
    def __init__(self, kv_client: KubeVirtClient):
        self.kv = kv_client

    def _map_status(self, vm_data: dict) -> VMStatus:
        status = vm_data.get("status", {})
        if status.get("ready"):
            return VMStatus.running
        print_status = status.get("printableStatus", "").lower()
        status_map = {
            "running": VMStatus.running,
            "stopped": VMStatus.stopped,
            "starting": VMStatus.starting,
            "stopping": VMStatus.stopping,
            "migrating": VMStatus.migrating,
            "paused": VMStatus.paused,
            "provisioning": VMStatus.provisioning,
            "errorimagepull": VMStatus.error,
            "crashloopbackoff": VMStatus.error,
        }
        return status_map.get(print_status, VMStatus.unknown)

    def _derive_health(self, vm_status: VMStatus, vmi_data: dict | None) -> HealthStatus:
        if vm_status == VMStatus.stopped:
            return HealthStatus.unknown
        if vm_status == VMStatus.error:
            return HealthStatus.critical
        if vm_status == VMStatus.running:
            conditions = (vmi_data or {}).get("status", {}).get("conditions", [])
            has_agent = any(
                c.get("type") == "AgentConnected" and c.get("status") == "True"
                for c in conditions
            )
            return HealthStatus.healthy if has_agent else HealthStatus.unknown
        return HealthStatus.unknown

    def _vm_from_k8s(self, vm_data: dict, vmi_data: dict | None = None) -> VM:
        metadata = vm_data.get("metadata", {})
        spec = vm_data.get("spec", {})
        template_spec = spec.get("template", {}).get("spec", {})
        domain = template_spec.get("domain", {})
        resources = domain.get("resources", {}).get("requests", {})

        cpu_cores = domain.get("cpu", {}).get("cores", 1)
        memory_str = resources.get("memory", "1Gi")
        memory_mb = self._parse_memory(memory_str)

        vm_status = self._map_status(vm_data)

        ip_addresses = []
        node = None
        if vmi_data:
            for iface in vmi_data.get("status", {}).get("interfaces", []):
                if iface.get("ipAddress"):
                    ip_addresses.append(iface["ipAddress"])
            node = vmi_data.get("status", {}).get("nodeName")

        return VM(
            name=metadata.get("name", ""),
            namespace=metadata.get("namespace", ""),
            created_at=metadata.get("creationTimestamp"),
            labels=metadata.get("labels", {}),
            annotations=metadata.get("annotations", {}),
            status=vm_status,
            health=self._derive_health(vm_status, vmi_data),
            compute=VMCompute(cpu_cores=cpu_cores, memory_mb=memory_mb),
            ip_addresses=ip_addresses,
            node=node,
            run_strategy=spec.get("runStrategy", "RerunOnFailure"),
            os_type=metadata.get("labels", {}).get("kubevmui.io/os-type"),
        )

    def _parse_memory(self, mem_str: str) -> int:
        if isinstance(mem_str, int):
            return mem_str // (1024 * 1024)
        mem_str = str(mem_str)
        if mem_str.endswith("Gi"):
            return int(float(mem_str[:-2]) * 1024)
        if mem_str.endswith("Mi"):
            return int(float(mem_str[:-2]))
        if mem_str.endswith("G"):
            return int(float(mem_str[:-1]) * 1024)
        return int(mem_str) // (1024 * 1024)

    def list_vms(self, namespace: str) -> dict:
        vms_data = self.kv.list_vms(namespace)
        vmis_data = self.kv.list_vmis(namespace)
        vmi_map = {
            vmi["metadata"]["name"]: vmi
            for vmi in vmis_data
        }

        items = []
        for vm_data in vms_data:
            name = vm_data["metadata"]["name"]
            vmi = vmi_map.get(name)
            items.append(self._vm_from_k8s(vm_data, vmi))

        return {"items": items, "total": len(items)}

    def get_vm(self, namespace: str, name: str) -> VM | None:
        vm_data = self.kv.get_vm(namespace, name)
        if not vm_data:
            return None
        vmi_data = self.kv.get_vmi(namespace, name)
        return self._vm_from_k8s(vm_data, vmi_data)

    def create_vm(self, request: VMCreate) -> VM:
        body = self._build_vm_manifest(request)
        result = self.kv.create_vm(request.namespace, body)
        return self._vm_from_k8s(result)

    def delete_vm(self, namespace: str, name: str) -> None:
        self.kv.delete_vm(namespace, name)

    def vm_action(self, namespace: str, name: str, action: str) -> dict:
        return self.kv.vm_action(namespace, name, action)

    def _build_vm_manifest(self, request: VMCreate) -> dict:
        disks = []
        volumes = []
        for i, disk in enumerate(request.disks):
            disk_name = disk.name or f"disk-{i}"
            disks.append({
                "name": disk_name,
                "disk": {"bus": disk.bus},
                **({"bootOrder": disk.boot_order} if disk.boot_order else {}),
            })
            volumes.append({
                "name": disk_name,
                "dataVolume": {"name": disk_name},
            })

        if request.cloud_init_user_data:
            disks.append({"name": "cloudinit", "disk": {"bus": "virtio"}})
            volumes.append({
                "name": "cloudinit",
                "cloudInitNoCloud": {"userData": request.cloud_init_user_data},
            })

        interfaces = []
        networks = []
        for i, net in enumerate(request.networks):
            iface_name = f"nic-{i}"
            interfaces.append({"name": iface_name, "bridge": {}})
            networks.append({
                "name": iface_name,
                "multus": {"networkName": net.network_profile},
            })

        if not interfaces:
            interfaces.append({"name": "default", "masquerade": {}})
            networks.append({"name": "default", "pod": {}})

        return {
            "apiVersion": "kubevirt.io/v1",
            "kind": "VirtualMachine",
            "metadata": {
                "name": request.name,
                "namespace": request.namespace,
                "labels": {
                    **request.labels,
                    **({"kubevmui.io/os-type": request.os_type} if request.os_type else {}),
                },
            },
            "spec": {
                "runStrategy": request.run_strategy,
                "template": {
                    "spec": {
                        "domain": {
                            "cpu": {"cores": request.compute.cpu_cores},
                            "resources": {
                                "requests": {"memory": f"{request.compute.memory_mb}Mi"},
                            },
                            "devices": {
                                "disks": disks,
                                "interfaces": interfaces,
                            },
                        },
                        "volumes": volumes,
                        "networks": networks,
                    },
                },
            },
        }
```

```python
from kubernetes import client
from kubernetes.client import ApiException

from app.models.common import NetworkType
from app.models.network_profile import NetworkProfile, NetworkProfileCreate


class NetworkService:
    NAD_GROUP = "k8s.cni.cncf.io"
    NAD_VERSION = "v1"
    NAD_PLURAL = "network-attachment-definitions"

    def __init__(self, api_client: client.ApiClient):
        self.custom_api = client.CustomObjectsApi(api_client)

    def list_profiles(self, namespace: str) -> dict:
        try:
            result = self.custom_api.list_namespaced_custom_object(
                group=self.NAD_GROUP,
                version=self.NAD_VERSION,
                namespace=namespace,
                plural=self.NAD_PLURAL,
            )
        except ApiException:
            return {"items": [], "total": 0}

        items = [self._nad_to_profile(nad) for nad in result.get("items", [])]
        return {"items": items, "total": len(items)}

    def get_profile(self, namespace: str, name: str) -> NetworkProfile | None:
        try:
            nad = self.custom_api.get_namespaced_custom_object(
                group=self.NAD_GROUP,
                version=self.NAD_VERSION,
                namespace=namespace,
                plural=self.NAD_PLURAL,
                name=name,
            )
            return self._nad_to_profile(nad)
        except ApiException as e:
            if e.status == 404:
                return None
            raise

    def create_profile(self, request: NetworkProfileCreate) -> NetworkProfile:
        nad_body = self._profile_to_nad(request)
        result = self.custom_api.create_namespaced_custom_object(
            group=self.NAD_GROUP,
            version=self.NAD_VERSION,
            namespace=request.namespace,
            plural=self.NAD_PLURAL,
            body=nad_body,
        )
        return self._nad_to_profile(result)

    def delete_profile(self, namespace: str, name: str) -> None:
        self.custom_api.delete_namespaced_custom_object(
            group=self.NAD_GROUP,
            version=self.NAD_VERSION,
            namespace=namespace,
            plural=self.NAD_PLURAL,
            name=name,
        )

    def _nad_to_profile(self, nad: dict) -> NetworkProfile:
        metadata = nad.get("metadata", {})
        annotations = metadata.get("annotations", {})
        import json
        config_str = nad.get("spec", {}).get("config", "{}")
        try:
            config = json.loads(config_str) if isinstance(config_str, str) else config_str
        except json.JSONDecodeError:
            config = {}

        net_type = config.get("type", "bridge")
        type_map = {"bridge": NetworkType.bridge, "sriov": NetworkType.sriov}

        return NetworkProfile(
            name=metadata.get("name", ""),
            namespace=metadata.get("namespace", ""),
            created_at=metadata.get("creationTimestamp"),
            labels=metadata.get("labels", {}),
            annotations=annotations,
            display_name=annotations.get("kubevmui.io/display-name", metadata.get("name", "")),
            description=annotations.get("kubevmui.io/description", ""),
            network_type=type_map.get(net_type, NetworkType.bridge),
            vlan_id=config.get("vlan"),
            dhcp_enabled=annotations.get("kubevmui.io/dhcp", "true") == "true",
            subnet=annotations.get("kubevmui.io/subnet"),
            gateway=annotations.get("kubevmui.io/gateway"),
        )

    def _profile_to_nad(self, request: NetworkProfileCreate) -> dict:
        import json
        config = {
            "cniVersion": "0.3.1",
            "type": request.network_type.value if request.network_type != NetworkType.sriov else "sriov",
        }
        if request.vlan_id:
            config["vlan"] = request.vlan_id

        return {
            "apiVersion": f"{self.NAD_GROUP}/{self.NAD_VERSION}",
            "kind": "NetworkAttachmentDefinition",
            "metadata": {
                "name": request.name,
                "namespace": request.namespace,
                "annotations": {
                    "kubevmui.io/display-name": request.display_name,
                    "kubevmui.io/description": request.description,
                    "kubevmui.io/dhcp": str(request.dhcp_enabled).lower(),
                    **({"kubevmui.io/subnet": request.subnet} if request.subnet else {}),
                    **({"kubevmui.io/gateway": request.gateway} if request.gateway else {}),
                },
            },
            "spec": {"config": json.dumps(config)},
        }
```

```python
# backend/app/services/storage_service.py
from kubernetes import client
from kubernetes.client import ApiException

from app.models.disk import Disk, DiskCreate


class StorageService:
    CDI_GROUP = "cdi.kubevirt.io"
    CDI_VERSION = "v1beta1"

    def __init__(self, api_client: client.ApiClient):
        self.api_client = api_client
        self.core_api = client.CoreV1Api(api_client)
        self.storage_api = client.StorageV1Api(api_client)
        self.custom_api = client.CustomObjectsApi(api_client)

    def _get_storage_class_map(self) -> dict[str, str]:
        """Map StorageClass names to human-readable tier labels."""
        try:
            scs = self.storage_api.list_storage_class()
            result = {}
            for sc in scs.items:
                annotations = sc.metadata.annotations or {}
                label = annotations.get("kubevmui.io/tier-label", sc.metadata.name)
                result[sc.metadata.name] = label
            return result
        except ApiException:
            return {}

    def list_disks(self, namespace: str) -> dict:
        try:
            pvcs = self.core_api.list_namespaced_persistent_volume_claim(namespace)
        except ApiException:
            return {"items": [], "total": 0}

        sc_map = self._get_storage_class_map()
        items = []
        for pvc in pvcs.items:
            storage = pvc.spec.resources.requests.get("storage", "0Gi")
            size_gb = self._parse_size(storage)
            sc_name = pvc.spec.storage_class_name or "default"

            items.append(Disk(
                name=pvc.metadata.name,
                namespace=pvc.metadata.namespace,
                created_at=pvc.metadata.creation_timestamp,
                labels=pvc.metadata.labels or {},
                annotations=pvc.metadata.annotations or {},
                size_gb=size_gb,
                performance_tier=sc_map.get(sc_name, sc_name),
                storage_class=sc_name,
                access_mode=pvc.spec.access_modes[0] if pvc.spec.access_modes else "ReadWriteOnce",
                volume_mode=pvc.spec.volume_mode or "Filesystem",
                status=pvc.status.phase or "Pending",
                attached_vm=(pvc.metadata.annotations or {}).get("kubevmui.io/attached-vm"),
            ))
        return {"items": items, "total": len(items)}

    def create_disk(self, request: DiskCreate) -> Disk:
        sc_map = self._get_storage_class_map()
        sc_name = None
        for name, label in sc_map.items():
            if label == request.performance_tier or name == request.performance_tier:
                sc_name = name
                break
        if not sc_name:
            sc_name = request.performance_tier

        body = client.V1PersistentVolumeClaim(
            metadata=client.V1ObjectMeta(
                name=request.name,
                namespace=request.namespace,
                labels=request.labels,
            ),
            spec=client.V1PersistentVolumeClaimSpec(
                access_modes=["ReadWriteOnce"],
                storage_class_name=sc_name,
                resources=client.V1VolumeResourceRequirements(
                    requests={"storage": f"{request.size_gb}Gi"},
                ),
            ),
        )
        result = self.core_api.create_namespaced_persistent_volume_claim(request.namespace, body)
        return Disk(
            name=result.metadata.name,
            namespace=result.metadata.namespace,
            created_at=result.metadata.creation_timestamp,
            size_gb=request.size_gb,
            performance_tier=sc_map.get(sc_name, sc_name),
            storage_class=sc_name,
            status="Pending",
        )

    def delete_disk(self, namespace: str, name: str) -> None:
        self.core_api.delete_namespaced_persistent_volume_claim(name, namespace)

    def _parse_size(self, size_str: str) -> int:
        size_str = str(size_str)
        if size_str.endswith("Gi"):
            return int(float(size_str[:-2]))
        if size_str.endswith("Ti"):
            return int(float(size_str[:-2]) * 1024)
        if size_str.endswith("Mi"):
            return max(1, int(float(size_str[:-2]) / 1024))
        return int(size_str) // (1024 ** 3)
```

```python
# backend/app/services/template_service.py
import json

from kubernetes import client
from kubernetes.client import ApiException

from app.models.template import Template, TemplateCreate
from app.models.vm import VMCompute, VMDiskRef, VMNetworkRef


TEMPLATE_LABEL = "kubevmui.io/type"
TEMPLATE_LABEL_VALUE = "template"


class TemplateService:
    def __init__(self, api_client: client.ApiClient):
        self.core_api = client.CoreV1Api(api_client)

    def list_templates(self, namespace: str) -> dict:
        try:
            cms = self.core_api.list_namespaced_config_map(
                namespace,
                label_selector=f"{TEMPLATE_LABEL}={TEMPLATE_LABEL_VALUE}",
            )
        except ApiException:
            return {"items": [], "total": 0}

        items = [self._cm_to_template(cm) for cm in cms.items]
        return {"items": items, "total": len(items)}

    def get_template(self, namespace: str, name: str) -> Template | None:
        try:
            cm = self.core_api.read_namespaced_config_map(f"tpl-{name}", namespace)
            if (cm.metadata.labels or {}).get(TEMPLATE_LABEL) != TEMPLATE_LABEL_VALUE:
                return None
            return self._cm_to_template(cm)
        except ApiException as e:
            if e.status == 404:
                return None
            raise

    def create_template(self, request: TemplateCreate) -> Template:
        spec_data = {
            "display_name": request.display_name,
            "description": request.description,
            "category": request.category,
            "os_type": request.os_type,
            "compute": request.compute.model_dump(),
            "disks": [d.model_dump() for d in request.disks],
            "networks": [n.model_dump() for n in request.networks],
            "cloud_init_user_data": request.cloud_init_user_data,
        }
        cm = client.V1ConfigMap(
            metadata=client.V1ObjectMeta(
                name=f"tpl-{request.name}",
                namespace=request.namespace,
                labels={TEMPLATE_LABEL: TEMPLATE_LABEL_VALUE},
            ),
            data={"spec": json.dumps(spec_data)},
        )
        result = self.core_api.create_namespaced_config_map(request.namespace, cm)
        return self._cm_to_template(result)

    def delete_template(self, namespace: str, name: str) -> None:
        self.core_api.delete_namespaced_config_map(f"tpl-{name}", namespace)

    def _cm_to_template(self, cm) -> Template:
        spec_str = (cm.data or {}).get("spec", "{}")
        spec = json.loads(spec_str)
        compute_data = spec.get("compute", {"cpu_cores": 1, "memory_mb": 1024})

        return Template(
            name=cm.metadata.name.removeprefix("tpl-"),
            namespace=cm.metadata.namespace,
            created_at=cm.metadata.creation_timestamp,
            labels=cm.metadata.labels or {},
            display_name=spec.get("display_name", cm.metadata.name),
            description=spec.get("description", ""),
            category=spec.get("category", "custom"),
            os_type=spec.get("os_type"),
            compute=VMCompute(**compute_data),
            disks=[VMDiskRef(**d) for d in spec.get("disks", [])],
            networks=[VMNetworkRef(**n) for n in spec.get("networks", [])],
            cloud_init_user_data=spec.get("cloud_init_user_data"),
        )
```

```python
# backend/app/ws/vnc_proxy.py
import asyncio

from fastapi import APIRouter, WebSocket, WebSocketDisconnect

from app.core.cluster_manager import ClusterManager

router = APIRouter()


@router.websocket("/ws/vnc/{cluster}/{namespace}/{vm}")
async def vnc_proxy(
    websocket: WebSocket,
    cluster: str,
    namespace: str,
    vm: str,
):
    await websocket.accept()
    cm: ClusterManager = websocket.app.state.cluster_manager
    api_client = cm.get_api_client(cluster)

    if not api_client:
        await websocket.close(code=4004, reason="Cluster not found")
        return

    # Build the VNC subresource WebSocket URL
    host = api_client.configuration.host
    ws_url = (
        f"{host.replace('https://', 'wss://').replace('http://', 'ws://')}"
        f"/apis/subresources.kubevirt.io/v1"
        f"/namespaces/{namespace}/virtualmachineinstances/{vm}/vnc"
    )

    try:
        import websockets
        ssl_context = None
        if api_client.configuration.ssl_ca_cert:
            import ssl
            ssl_context = ssl.create_default_context(cafile=api_client.configuration.ssl_ca_cert)

        headers = {}
        if api_client.configuration.api_key:
            token = api_client.configuration.api_key.get("authorization", "")
            if token:
                headers["Authorization"] = token

        async with websockets.connect(ws_url, ssl=ssl_context, additional_headers=headers) as upstream:
            async def forward_to_upstream():
                try:
                    while True:
                        data = await websocket.receive_bytes()
                        await upstream.send(data)
                except WebSocketDisconnect:
                    pass

            async def forward_to_client():
                try:
                    async for message in upstream:
                        if isinstance(message, bytes):
                            await websocket.send_bytes(message)
                        else:
                            await websocket.send_text(message)
                except Exception:
                    pass

            await asyncio.gather(forward_to_upstream(), forward_to_client())
    except Exception as e:
        await websocket.close(code=4000, reason=str(e)[:120])
```

```python
# backend/app/api/routes/dashboard.py
from fastapi import APIRouter, Depends, HTTPException

from app.api.deps import get_cluster_manager, get_current_user
from app.core.cluster_manager import ClusterManager
from app.core.k8s_client import KubeVirtClient
from app.models.auth import UserInfo

router = APIRouter(prefix="/api/v1/clusters/{cluster}", tags=["dashboard"])


def get_dashboard_stats(kv: KubeVirtClient) -> dict:
    namespaces = kv.list_namespaces()
    total_vms = 0
    running = 0
    stopped = 0
    errors = 0

    for ns in namespaces:
        try:
            vms = kv.list_vms(ns)
            vmis = kv.list_vmis(ns)
            vmi_names = {v["metadata"]["name"] for v in vmis}
            total_vms += len(vms)
            for vm in vms:
                name = vm["metadata"]["name"]
                status = vm.get("status", {})
                if name in vmi_names:
                    if status.get("ready"):
                        running += 1
                    else:
                        ps = status.get("printableStatus", "").lower()
                        if "error" in ps or "crash" in ps:
                            errors += 1
                        else:
                            running += 1
                else:
                    stopped += 1
        except Exception:
            continue

    nodes = kv.list_nodes()

    return {
        "total_vms": total_vms,
        "running_vms": running,
        "stopped_vms": stopped,
        "error_vms": errors,
        "node_count": len(nodes),
    }


@router.get("/dashboard")
async def dashboard(
    cluster: str,
    cm: ClusterManager = Depends(get_cluster_manager),
    user: UserInfo = Depends(get_current_user),
):
    api_client = cm.get_api_client(cluster)
    if not api_client:
        raise HTTPException(status_code=404, detail=f"Cluster '{cluster}' not found")
    kv = KubeVirtClient(api_client)
    return get_dashboard_stats(kv)
```

```typescript
// frontend/src/hooks/useAuth.ts
import { useState, useCallback } from 'react'
import apiClient from '@/lib/api-client'

interface User {
  username: string
  groups: string[]
}

export function useAuth() {
  const [user, setUser] = useState<User | null>(() => {
    const stored = localStorage.getItem('kubevmui_user')
    return stored ? JSON.parse(stored) : null
  })

  const login = useCallback(async (token: string) => {
    const { data } = await apiClient.post('/auth/login', { token })
    const userData: User = { username: data.username, groups: data.groups }
    localStorage.setItem('kubevmui_user', JSON.stringify(userData))
    setUser(userData)
    return userData
  }, [])

  const logout = useCallback(async () => {
    await apiClient.post('/auth/logout')
    localStorage.removeItem('kubevmui_user')
    setUser(null)
  }, [])

  return { user, login, logout, isAuthenticated: !!user }
}
```

```typescript
import { useEffect, useRef } from 'react'
import RFB from '@novnc/novnc/lib/rfb'

interface VNCConsoleProps {
  cluster: string
  namespace: string
  vmName: string
}

export function VNCConsole({ cluster, namespace, vmName }: VNCConsoleProps) {
  const containerRef = useRef<HTMLDivElement>(null)
  const rfbRef = useRef<RFB | null>(null)

  useEffect(() => {
    if (!containerRef.current) return

    const protocol = window.location.protocol === 'https:' ? 'wss:' : 'ws:'
    const wsUrl = `${protocol}//${window.location.host}/ws/vnc/${cluster}/${namespace}/${vmName}`

    const rfb = new RFB(containerRef.current, wsUrl)
    rfb.scaleViewport = true
    rfb.resizeSession = true
    rfbRef.current = rfb

    return () => {
      rfb.disconnect()
      rfbRef.current = null
    }
  }, [cluster, namespace, vmName])

  return (
    <div
      ref={containerRef}
      style={{ width: '100%', height: '100%', background: '#000' }}
    />
  )
}
```

- [ ] **Step 3: Build serial console component**

```typescript
// frontend/src/components/console/SerialConsole.tsx
import { useEffect, useRef } from 'react'
import { Terminal } from 'xterm'
import { FitAddon } from '@xterm/addon-fit'
import 'xterm/css/xterm.css'

interface SerialConsoleProps {
  cluster: string
  namespace: string
  vmName: string
}

export function SerialConsole({ cluster, namespace, vmName }: SerialConsoleProps) {
  const terminalRef = useRef<HTMLDivElement>(null)

  useEffect(() => {
    if (!terminalRef.current) return

    const term = new Terminal({
      theme: {
        background: '#0a0a0b',
        foreground: '#e4e4e7',
        cursor: '#6366f1',
      },
      fontFamily: 'JetBrains Mono, monospace',
      fontSize: 14,
    })
    const fitAddon = new FitAddon()
    term.loadAddon(fitAddon)
    term.open(terminalRef.current)
    fitAddon.fit()

    const protocol = window.location.protocol === 'https:' ? 'wss:' : 'ws:'
    const wsUrl = `${protocol}//${window.location.host}/ws/console/${cluster}/${namespace}/${vmName}`
    const ws = new WebSocket(wsUrl)

    ws.onmessage = (event) => {
      term.write(event.data)
    }

    term.onData((data) => {
      ws.send(data)
    })

    const resizeObserver = new ResizeObserver(() => fitAddon.fit())
    resizeObserver.observe(terminalRef.current)

    return () => {
      ws.close()
      term.dispose()
      resizeObserver.disconnect()
    }
  }, [cluster, namespace, vmName])

  return <div ref={terminalRef} style={{ width: '100%', height: '100%' }} />
}
```
